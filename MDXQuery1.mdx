---1)TÊN CÁC MẶT GIỐNG
select {[DIMGIONG].[TEN GIONG].[TEN GIONG]} on COLUMNS
from [OLAP]
--2)DOANH THU TỪNG GIỐNG
SELECT [Measures].[TONGTIEN] ON COLUMNS,
[DIMGIONG].[TEN GIONG].[TEN GIONG] ON ROWS
FROM [OLAP]
--3)DOANH THU TỪNG NĂM 2019 2018 2020
SELECT {[DIMNAM].[NAM].&[2018],[NAM].[NAM].&[2019],[DIMNAM].[NAM].&[2020] }ON ROWS,
[Measures].[TONGTIEN] ON COLUMNS
FROM [OLAP]
--4)TRONG NĂM 2019, NHỮNG CỬA HÀNG ĐÃ LẤY SỐ LƯỢNG VẬT NUÔI 
SELECT NON EMPTY [DIMCUAHANG].[TEN CUA HANG].[TEN CUA HANG] ON COLUMNS, [DIMGIONG].[TEN GIONG].[TEN GIONG] ON ROWS
FROM [OLAP]
WHERE [DIMNAM].[NAM].&[2.019E3]
--5)TRONG 2019 , DOANH THU BÁN ĐƯỢC TỪNG GIỐNG
SELECT NON EMPTY [Measures].[TONGTIEN]ON COLUMNS, [DIMGIONG].[TEN GIONG].[TEN GIONG] ON ROWS
FROM [OLAP]
WHERE [DIMNAM].[NAM].&[2.019E3]
--6) SỐ LƯỢNG BÁN RA TẠI MỖI CỬA HÀNG TRONG năm 2019
SELECT NON EMPTY CROSSJOIN([DIMGIONG].[Hierarchy].[MA GIONG].MEMBERS,[DIMTRANGTRAI].[TEN TRANGTRAI].MEMBERS ) ON ROWS,[DIMNAM].[NAM].[NAM].&[2020] ON COLUMNS
FROM[OLAP]

--7) BÁO CÁO SỐ LƯỢNG CỦA MỖI TRANG TRẠI THEO TỪNG NĂM

FROM[OLAP]
--8)BÁO CÁO DOANH THU CỦA MỖI TRANG TRẠI THEO TỪNG NĂM
SELECT NON EMPTY CROSSJOIN( [DIMTRANGTRAI].[TEN TRANGTRAI].[TEN TRANGTRAI].MEMBERS,[DIMNAM].[NAM].[NAM]) ON ROWS,NON EMPTY[Measures].[TONGTIEN] ON COLUMNS
FROM[OLAP]
--9)Báo cáo số lượng cửa hàng trong năm 2020 lấy bao nhiêu sản lượng của trang trại và giống nào
select non empty CROSSJOIN([DIMGIONG].[Hierarchy].[MA GIONG].MEMBERS,[DIMTRANGTRAI].[TEN TRANGTRAI].MEMBERS )  on rows, CROSSJOIN( [DIMCUAHANG].[TEN CUA HANG].[TEN CUA HANG],[DIMNAM].[NAM].[NAM].&[2020]) 
on columns from [OLAP]
--10) 3 tháng có doanh thu cao nhất trong 2019

from [OLAP]
